double f(double x) {
	return (-10 + 1 * x) / (2 + 0 * x);
}

double Simpson() {
	double a = 0, b = 6; //limites de la integral a <= b
	const int N = 1000000;//valor predefinido
	double s = 0;
	double h = (b - a) / N;
	for (int i = 0; i <= N; ++i) {
		double x = a + h * i;
		//f(x) es la funcion a la que se le quiere hallar la integral
		double y = f(x);
		s += y * ((i == 0 || i == N) ? 1 : ((i & 1) == 0) ? 2 : 4);
	}
	s *= h / 3;
	return s;
}
