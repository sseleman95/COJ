typedef long long ll;

void DigitCount(int n, ll *sol) {
	ll aux = n, sum = 0, p = 1, d;
	while (aux) {
		d = aux % 10, aux /= 10;
		sol[d] += ((n % p) + 1);
		for (int i = 0; i < d; i++)
			sol[i] += p;
		for (int i = 0; i < 10; i++)
			sol[i] += sum * d;
		sol[0] -= p;
		sum = p + 10 * sum;
		p *= 10;
	}
}
