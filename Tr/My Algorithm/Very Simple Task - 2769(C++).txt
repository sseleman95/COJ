using namespace std;
typedef long long ll;
const int MOD = 1000000007;

ll sum[1000001];

ll modulo(ll a, ll b) {
	ll x = 1, y = a;
	while (b > 0) {
		if (b % 2 == 1) {
			x = (x * y) % MOD;
		}
		y = (y * y) % MOD;
		b /= 2;
	}
	return x % MOD;
}

int main() {
	//freopen("f", "r", stdin);
	for (ll i = 0, j = 0; j <= 1000000; ++i, ++j) {
		sum[j] = modulo(2, i);
		sum[j] += sum[j - 1];
		sum[j] %= MOD;
	}
	int n;
	while (1) {
		scanf("%d", &n);
		if (!n)
			break;
		cout << sum[n] << endl;
	}
}
