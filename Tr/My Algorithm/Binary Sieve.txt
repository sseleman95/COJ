
const int MAXSIZE = 100000000 / 2 / 8 + 1;
char p[MAXSIZE];

//(p[i>>3]&(1<<(i&7))) == 0 if 2*i + 1 is prime

int getTheNumber(int n) {
	int i, j, nr = 1;
	for (i = 1; ((i * i) << 1) + (i << 1) <= n; i += 1) {
		if ((p[i >> 3] & (1 << (i & 7))) == 0) {
			for (j = ((i * i) << 1) + (i << 1); (j << 1) + 1 <= n;
					j += (i << 1) + 1) {
				p[j >> 3] |= (1 << (j & 7));
			}
		}
	}
	for (i = 1; 2 * i + 1 <= n; ++i)
		if ((p[i >> 3] & (1 << (i & 7))) == 0)
			nr++;
	return nr;
}
