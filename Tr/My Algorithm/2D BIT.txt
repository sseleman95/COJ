int tree[MAXN + 1][MAXN + 1];

void update(int x, int y, int val, int N) {
	int y1;
	while (x <= N) {
		y1 = y;
		while (y1 <= N) {
			tree[x][y1] = ((tree[x][y1] + val % MOD) + MOD) % MOD;
			y1 += (y1 & -y1);
		}
		x += (x & -x);
	}
}

int query(int x, int y) {
	int sum = 0, y1;
	while (x > 0) {
		y1 = y;
		while (y1 > 0) {
			sum = (sum + tree[x][y1]) % MOD;
			y1 -= (y1 & -y1);
		}
		x -= (x & -x);
	}
	return sum;
}
