using namespace std;

char text1[1005], text2[1005];

int main() {
	//freopen("f", "r", stdin);
	int n;
	scanf("%d", &n);
	n %= 2;
	scanf("%s\n%s", text1, text2);
	bool ok = 1;
	if (n)
		for (int i = 0; text1[i]; ++i) {
			if (text1[i] == text2[i]) {
				ok = 0;
				break;
			}
		}
	else
		for (int i = 0; text1[i]; ++i) {
			if (text1[i] != text2[i]) {
				ok = 0;
				break;
			}
		}
	if (ok)
		puts("Deletion succeeded");
	else
		puts("Deletion failed");
}
