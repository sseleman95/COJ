using namespace std;

int nums[1000], added[1000 * 1001 / 2];

int main() {
	//freopen("f", "r", stdin);
	int max_val;
	scanf("%d", &max_val);
	int n;
	scanf("%d", &n);
	for (int i = 0; i < n; ++i) {
		scanf("%d", &nums[i]);
	}
	int l = 0;
	for (int i = 0; i < n; ++i)
		for (int j = i + 1; j < n; ++j)
			added[l++] = nums[i] + nums[j];
	sort(added, added + l);
	int cnt = 0;
	for (int i = 0; i < n; ++i) {
		if (nums[i] <= max_val) {
			int diff = max_val - nums[i];
			int idx = lower_bound(added, added + l, diff) - added;
			if ((idx == 0 && added[idx] == diff) || idx)
				cnt++;
		}
	}
	printf("%d\n", cnt);
}
