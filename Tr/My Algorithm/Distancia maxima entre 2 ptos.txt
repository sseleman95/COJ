double diameter(Pol pt) {//24, 1 
   int is=0,js=0, n=pt.size();
   FAB(i,1,n){
      if(pt[i].Y >pt[is].Y) is=i;
      if(pt[i].Y <pt[js].Y) js=i;
   }
   double maxd=norm(pt[is]-pt[js]);
   int i,maxi,j,maxj;
   i = maxi = is;  j = maxj = js;
   do {
     if(cross(pt[(i+1)%n]-pt[i],
             pt[(j+1)%n]-pt[j])>=0) 
       j=(j+1)%n; else i=(i+1)%n;
      if (norm(pt[i]-pt[j])>maxd){
         maxd =norm(pt[i]-pt[j]);
         maxi=i; maxj=j;
  } }while(i!=is || j!=js);
   return maxd;
}
