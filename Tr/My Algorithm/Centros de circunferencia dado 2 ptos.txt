//a, b son las coordenadas del pto1(x, y)
//c, d son las coordenadas del pto2(x, y)
//retorna cero si la distancia entre los dos ptos es mayor que el diametro permitido
//r es el radio para calcular el diametro permitido
double x1, Y1, x2, y2;
int centre(double a, double b, double c, double d, double r) {
	double rise, run;
	double chordlen;
	double tantheta;
	chordlen = hypot(a - c, b - d);
	if (chordlen > 2 * r)
		return 0;
	tantheta = sqrt(r * r * 4 - chordlen * chordlen) / (chordlen);
	run = (a - c) / 2;
	rise = (b - d) / 2;
	x1 = (a + c) / 2 + rise * tantheta;
	Y1 = (b + d) / 2 + -run * tantheta;
	x2 = (a + c) / 2 + -rise * tantheta;
	y2 = (b + d) / 2 + run * tantheta;
	return 1;
}
