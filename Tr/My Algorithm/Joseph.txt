//para un k mayor O(n)
int joseph(int n, int k) {
	int res = 0;
	for (int i = 1; i <= n; ++i)
		res = (res + k) % i;
	return res + 1;
}

//para un k pequeño k log n
ll joseph1(ll n, ll k) {
	if (n == 1)
		return 0;
	if (k == 1)
		return n - 1;
	if (k > n)
		return (joseph1(n - 1, k) + k) % n;
	ll cnt = n / k;
	ll res = joseph1(n - cnt, k);
	res -= n % k;
	if (res < 0)
		res += n;
	else
		res += res / (k - 1);
	return res;
}
